from __future__ import annotations
import sys
import os

# ГўвЂќв‚¬ГўвЂќв‚¬ Fix: pytest rootdir detection adds D:\Projetos to sys.path[0], which
# causes Python to find finanalytics_ai as a namespace package pointing to
# D:\Projetos\finanalytics_ai (no `domain` there). We must:
# 1. Remove any path that shadows the real package (dirs containing
#    finanalytics_ai/ without __init__.py = namespace package hijack)
# 2. Clear sys.modules cache of any already-imported namespace entries
# 3. Ensure src/ is first

_SRC = os.path.normpath(
    os.path.join(os.path.dirname(os.path.abspath(__file__)), "src")
)

# Step 1: purge shadowing paths
sys.path = [
    p for p in sys.path
    if not (
        os.path.isdir(os.path.join(p, "finanalytics_ai"))
        and not os.path.isfile(os.path.join(p, "finanalytics_ai", "__init__.py"))
    )
]

# Step 2: clear stale namespace package cache
for key in list(sys.modules.keys()):
    if "finanalytics_ai" in key:
        del sys.modules[key]

# Step 3: ensure src is first

class Currency(StrEnum):
    BRL = "BRL"; USD = "USD"; EUR = "EUR"

@dataclass(frozen=True)
class Money:
    """Valor monetГѓВЎrio com moeda. Usa Decimal para precisГѓВЈo financeira."""
    amount: Decimal
    currency: Currency = Currency.BRL
    _PLACES: ClassVar[Decimal] = Decimal("0.01")

    def __post_init__(self) -> None:
        object.__setattr__(self, "amount", self.amount.quantize(self._PLACES, ROUND_HALF_UP))

    @classmethod
    def of(cls, value: str | float | int | Decimal, currency: Currency = Currency.BRL) -> "Money":
        try:
            return cls(amount=Decimal(str(value)), currency=currency)
        except InvalidOperation as exc:
            raise ValueError(f"Valor invГѓВЎlido: {value!r}") from exc

    def __add__(self, other: "Money") -> "Money":
        self._check(other); return Money(self.amount + other.amount, self.currency)
    def __sub__(self, other: "Money") -> "Money":
        self._check(other); return Money(self.amount - other.amount, self.currency)
    def __mul__(self, factor: Decimal | int | float) -> "Money":
        return Money(self.amount * Decimal(str(factor)), self.currency)
    def __lt__(self, other: "Money") -> bool:
        self._check(other); return self.amount < other.amount
    def __le__(self, other: "Money") -> bool:
        self._check(other); return self.amount <= other.amount
    def is_positive(self) -> bool: return self.amount > Decimal("0")
    def is_zero(self) -> bool: return self.amount == Decimal("0")
    def _check(self, other: "Money") -> None:
        if self.currency != other.currency:
            raise ValueError(f"Moedas diferentes: {self.currency} vs {other.currency}")
    def __repr__(self) -> str: return f"Money({self.amount} {self.currency})"
